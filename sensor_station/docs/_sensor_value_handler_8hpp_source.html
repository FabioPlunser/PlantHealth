133 str->light_intensity =

143 uint16_t convertToGATT_soilHumidity(uint16_t humidity) {
144 DEBUG_PRINT_POS(4, "\n");
145 DEBUG_PRINTF_POS(2, "Soil humidity = %u\n", humidity);
146 // Calibration values
147 static uint16_t valueHigh = 300;
148 static uint16_t valueLow = 950;
149 float calculation = 100 - ((humidity - valueHigh) /
150 (float) (valueLow - valueHigh) * 100);
151
152 DEBUG_PRINTF_POS(2, "Calculated = %lf\n", calculation);
153 if (calculation < 0 || calculation > 100) {
154 return convertToGATT_soilHumidity_notKnown();
155 }
156 return uint16_t(calculation * 100);
157 }

262 uint16_t luminosityFromVoltage(uint16_t measured) {
263 DEBUG_PRINT_POS(4, "\n");

266 float Vout =
267 float(measured) / 1023 * Vin; // Convert analog to voltage

270 return 500 / (diodaResistance / 1000
